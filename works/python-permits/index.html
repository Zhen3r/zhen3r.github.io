In [8]:
permits_CnR = (
    permits_CnR
    .groupby(["zip","commercialorresidential"])
    .agg({"commercialorresidential":"count"})
    .rename({"commercialorresidential":"totalNumber"},axis=1)
    .reset_index()
    .rename({"commercialorresidential":"type"},axis=1)
    .pivot(index='zip', columns='type', values='totalNumber')
    .reset_index().fillna(0)
)
permits_CnR["totalNumber"] = permits_CnR["COMMERCIAL"]+permits_CnR["RESIDENTIAL"]
permits_CnR = permits_CnR.sort_values("totalNumber",ascending=False).reset_index(drop=True)

brush = alt.selection_interval(encodings=['x'],empty='all')
color_scale = alt.Color("PermitType:O",scale=alt.Scale(scheme="plasma"),legend=None)

time_plot = (alt.Chart()
    .mark_line(color="lightgrey")
    .mark_area(color="#db565d11")
    .encode(
        x=alt.X("Month:T",title="Date"),
        y="sum(PermitNumber):Q")
    .properties(
        width=800, height=70,
        title='Type of building permits in Philadelphia, 2016-2021')
)

background = time_plot.add_selection(brush)
selected = time_plot.transform_filter(brush).mark_line(color='#db565d')
#.mark_area(color="#FEF4F4")

bar_plot = (alt.Chart()
    .mark_bar()
    .encode(
        x=alt.X("PermitType:O",sort="-y"),
        y=alt.Y("sum(PermitNumber):Q"),
        color = color_scale,
        tooltip=["PermitType:O","sum(PermitNumber):Q"],)
    .properties(
        width=800, height=250)

)
